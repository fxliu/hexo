## 常规文件操作

```sh
function mymv()
{
    #echo $0
    #echo $*
    #echo $1
    #echo $2
    if [ -f $1 ]; then
        echo $1
        # 强制覆盖不提醒
        mv -f $1 ./
    fi
}
mymv ../myfile
```
