## 函数

```sh
# 函数
function mymv()
{
    #echo $0
    #echo $*
    #echo $1
    #echo $2
    if [ -f $1 ]; then
        echo $1
        # 强制覆盖不提醒
        mv -f $1 ./
    fi
}
mymv ../myfile
```

```sh
# 支持命令行参数
rm -rf ~/nfs/release/lib$1.so
cp ~/projects/$1/lib$1.so ~/nfs/release/lib$1.so
chmod 777 ~/nfs/release/lib$1.so
```
