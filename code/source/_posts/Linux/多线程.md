## 线程模板

```c++
#include <pthread.h>

template <class T>
class CMyThread
{
private:
  typedef void (T::*pcb)(LPVOID);

  T* m_t;
  void (T::*m_cb)(LPVOID);

  pthread_t m_thread;
  LPVOID m_cbParam;

  BOOL m_run;
  BOOL m_exit;
public:
  CMyThread() {
    m_t = NULL;
    m_cb = NULL;
    m_cbParam = NULL;
    m_thread = 0;
    m_run = FALSE;
    m_exit = FALSE;
  }
  CMyThread(T* t, pcb cb, LPVOID lpVoid) {
    m_t = t;
    m_cb = cb;
    m_cbParam = lpVoid;
    m_thread = 0;
    m_run = FALSE;
    m_exit = FALSE;
  }
  ~CMyThread() {};
private:
  static void* _RunOnce(LPVOID lpVoid) {
    CMyThread<T> *pThis = (CMyThread<T>*)lpVoid;
    pThis->Run();
    delete pThis;
    return 0;
  }
  static void* _Run(LPVOID lpVoid) {
    CMyThread<T> *pThis = (CMyThread<T>*)lpVoid;
    pThis->Run();
    return 0;
  }
  void Run() {
    if (m_t && m_cb) {
      (m_t->*m_cb)(m_cbParam);
    }
    m_run = FALSE;
  }
public:
  static void StartOnce(T* t, pcb cb, LPVOID lpVoid) {
    CMyThread<T> *pThread = new CMyThread<T>(t, cb, lpVoid);
    pThread->m_run = TRUE;
    pThread->m_exit = FALSE;
    pthread_t thread;
    pthread_create(&thread, NULL, _RunOnce, pThread);
    pthread_detach(thread);
  }
public:
  void Start(T* t, pcb cb, LPVOID lpVoid) {
    m_t = t;
    m_cb = cb;
    m_cbParam = lpVoid;
    m_run = TRUE;
    m_exit = FALSE;
    pthread_create(&m_thread, NULL, _Run, this);
  }
  void Stop() {
    m_exit = TRUE;
    if (m_thread)
    {
      pthread_join(m_thread, NULL);
      m_thread = 0;
    }
  }
  BOOL IsRun() {
    return m_run;
  }
  BOOL IsExit() {
    return m_exit;
  }
};

```
